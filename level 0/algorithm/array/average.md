## moyenne


            int somme = 0;
            int moyenne = 0;
            int[] tab = {1, 4, 6, 6, 3, 9, 3, 0, 3};
            
            //nombre d'elements
            int n = tab.length;
           
            //somme
            for (int j = 0; j < n; j++) {
                somme += tab[j];
            }
            //moyenne = somme/nombre
            moyenne = somme / n;
